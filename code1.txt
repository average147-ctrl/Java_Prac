#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    int n;
    
    cout << "Введите количество элементов в массиве: ";
    cin >> n;
    
    if (n <= 0) {
        cout << "Количество элементов должно быть положительным числом!" << endl;
        return 1;
    }
    
    vector<int> numbers;
    numbers.reserve(n);
    
    cout << "Введите " << n << " элементов массива:" << endl;
    
    for (int i = 0; i < n; ++i) {
        int num;
        cout << "Элемент " << i + 1 << ": ";
        cin >> num;
        numbers.push_back(num);
    }
    
    if (numbers.empty()) {
        cout << "Массив пуст!" << endl;
        return 1;
    }
    
    auto max_it = max_element(numbers.begin(), numbers.end());
    
    cout << "Максимальное значение: " << *max_it << endl;
    
    return 0;
}
